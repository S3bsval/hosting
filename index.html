// Crea un archivo llamado "server.js" y pega este código const http = require('http'); // Crear un servidor const server = http.createServer((req, res) => { res.statusCode = 200; // Código de respuesta: OK res.setHeader('Content-Type', 'text/plain'); // Encabezado de respuesta res.end('Hola Mundo\n'); // Contenido que se envía al cliente }); // Define el puerto y comienza a escuchar const port = 3000; server.listen(port, () => { console.log(`Servidor corriendo en http://localhost:${port}/`); });
